  select exists (
    select 1 from admin_users
    where id = auth.uid() and role = 'super-admin'
  );

create function log_admin_change() returns trigger
language plpgsql security definer as $$
begin
  insert into audit_log(actor_id, action, table_name, row_id, before_json, after_json)
  values(
    auth.uid(), TG_OP, TG_TABLE_NAME,
    coalesce(NEW.id, OLD.id),
    case when TG_OP = 'INSERT' then null else to_jsonb(OLD) end,
    case when TG_OP = 'DELETE' then null else to_jsonb(NEW) end
  );
  return coalesce(NEW, OLD);
end; $$;

function drivingGuideDailyCost(v: VehicleSpec, lang: LangRegion, grade: GuideGrade): number {

계산 정밀도 규칙
  • 중간 계산은 JavaScript Number (IEEE 754 double) 그대로 사용 — 통화 합산 ~20조 KRW까지 안전.
  • 최종 출력 시 Math.round로 1원 단위 반올림. USD는 0.01 단위.
  • 표시 포맷: Intl.NumberFormat('ko-KR') · Intl.NumberFormat('en-US', { currency: 'USD' }).
  • 음수 입력 차단 — react-hook-form + zod 스키마에서 z.number().nonnegative().

자기 비판 5건
  1. "비로그인 누구나" 가정이 트래픽 모델을 너무 단순화했다. 한국 인바운드 여행사·가이드가 "많이 사용할 것"이라는 대표 통찰은 정성 판단이며, 실제 채택률은 미검증. Phase 0 KPI "DAU ≥ 5"는 매우 보수적 수치. Phase 0 종료 후 실측이 가정과 5배 이상 어긋나면 사업 모델 재검토 필요.
  2. "각 인원별 단가"가 핵심 가치라면, 본 문서의 와이어프레임은 정량적 검증 부족. 결과 화면(P02)에 인원별 4구간 카드를 강조했으나, 실제 사용자가 "성인 178k" 한 숫자만 보는지·4구간을 모두 활용하는지는 유저 테스트 전엔 미지수.

S20

★ 금액 표시 규칙 (VAT 10% 포함 합계)

PRICING DISPLAY

2026-05-18 대표 확정 사항. 본 PWA의 모든 금액은 부가가치세(VAT) 10% 포함 합계로 표기한다. 별도 표기·세전 금액 표기는 금지. 본 규칙은 화면·엑셀·PDF·공유·API 응답·메모리 캐시 모두에 동일 적용.

규칙 7조

  1. 입력 시점 — 사용자가 user_library에 차량·가이드·식사 단가를 입력할 때 "VAT 10% 포함 합계"를 입력 라벨에 명시한다. 별도 VAT 입력 필드 없음.
  2. 저장 시점 — DB(user_library.payload·user_packages.computed_cache) 및 localStorage 모든 금액 컬럼은 VAT 포함 합계 그대로 저장. 별도 VAT 컬럼 없음.
  3. 계산 시점 — STEP 1~14의 모든 중간값·결과값은 VAT 포함 합계 기준. vatRate 변수는 호환성을 위해 유지하되 기본값 0, UI에서는 비노출.
  4. 출력 시점 — 화면·엑셀·PDF·QR·매직링크 모든 출력에 통화 단위로 "KRW (VAT 10% 포함 합계)" 명시. 시트 머리글 + 표 캡션 + PDF 푸터 3 위치.
  5. USD 병기 — USD 환산도 VAT 포함 합계 기준. finalUsd = finalKrw × exchangeRate.krwToUsd. 별도 세전 환산 없음.
  6. OTA·B2B 협상가 — 채널 수수료 차감 시 기준값도 VAT 포함 합계. profitPerAdult = netRevenue − tierCost의 양변 모두 VAT 포함 합계.
  7. 법적 표시 — 외국인 대상이지만 한국 부가세 환급(Tax Refund) 안내는 AI 보고서에 별도 첨부(STEP 14 결과 보강). 본 견적 단가 자체는 항상 VAT 포함.
